const botao = document.getElementById('btn-explicar'); const resultado = document.getElementById('resultado'); botao.addEventListener('click', async () => { const termo = document.getElementById('input-termo').value; // 1. Estado de carregando botao.disabled = true; resultado.innerText = "Pensando e gerando explicação fácil... Aguarde."; try { const response = await fetch('URL_DO_SEU_WORKER', { method: 'POST', body: JSON.stringify({ termo }), headers: { 'Content-Type': 'application/json' } }); if (!response.ok) throw new Error(); const data = await response.json(); // 2. Exibe o resultado resultado.innerText = data.resposta; } catch (error) { // 3. Tratamento de erro amigável resultado.innerText = "Ops! Ocorreu um erro ao falar com a IA. Tente novamente mais tarde."; } finally { // Restaura o botão botao.disabled = false; } });
